#version 330 core
layout (location = 0) in vec3 Lpos;
layout (location = 1) in vec3 Lcolor;
layout (location = 0) in vec3 LtexCords;
out vec4 SColor;
out vec2 STexCoord;
uniform mat4 transform;
uniform vec3 mcolor;
uniform mat4 tx_cord;
vec4 _txtrans;
void main()
{
    gl_Position = transform*vec4(Lpos.x,Lpos.y,Lpos.z, 1.0f);
    SColor = vec4(mcolor,1.0f);
    _txtrans=tx_cord*vec4(LtexCords.x,LtexCords.y,0,1.0f);
    STexCoord = vec2(_txtrans.x,_txtrans.y);
}
